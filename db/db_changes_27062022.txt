CREATE TABLE user_product_replacement_logs(
id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user_product_id BIGINT UNSIGNED,
`comment` TEXT,
`status` BOOLEAN DEFAULT 1,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE user_product_replacement_log_files(
id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user_product_replacement_log_id BIGINT UNSIGNED,
`file` VARCHAR(255),
`status` BOOLEAN DEFAULT 1,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE user_product_replacement_log_tags(
id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
user_product_replacement_log_id BIGINT UNSIGNED,
tag VARCHAR(100),
`status` BOOLEAN DEFAULT 1,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
